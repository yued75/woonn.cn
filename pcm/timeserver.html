🔗 在其他网站获取本页面的时间 JSON 
<iframe id="timeFrame" src="https://woonn.cn/pcm/timeserver.html" style="display:none;"></iframe>
<script>
    const iframe = document.getElementById('timeFrame');
    iframe.onload = () => {
        iframe.contentWindow.postMessage('getTime', '*');
    };
    window.addEventListener('message', (e) => {
        if (e.data && typeof e.data === 'object' && e.data.server_time) {
            console.log('获取到的服务器时间:', e.data);
            // e.data 包含 { server_time, local_time, offset_seconds, timestamp_unix_ms }
            alert('服务器时间: ' + e.data.server_time);
        }
    });
</script>
